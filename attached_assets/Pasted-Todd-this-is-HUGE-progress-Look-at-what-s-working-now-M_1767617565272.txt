function isValidYouTubeId(id: string | null | undefined): boolean {
  if (!id) return false;
  // YouTube IDs are exactly 11 characters, alphanumeric with - and _
  return /^[a-zA-Z0-9_-]{11}$/.test(id);
}

// 2. Build thumbnail URL with fallback
function getThumbnailUrl(youtubeId: string | null): string {
  if (youtubeId && isValidYouTubeId(youtubeId)) {
    // Try high quality first, browser will fallback if not available
    return `https://img.youtube.com/vi/${youtubeId}/hqdefault.jpg`;

}
