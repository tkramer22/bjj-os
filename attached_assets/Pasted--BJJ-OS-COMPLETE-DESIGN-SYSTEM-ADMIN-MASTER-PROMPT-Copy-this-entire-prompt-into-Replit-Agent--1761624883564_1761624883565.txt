```javascript
async function grantLifetimeAccess(email, reason, notes) {
  const user = await db.users.findOne({ email });
  
  if (!user) {
    // Create account automatically
    const newUser = await db.users.create({
      email,
      lifetimeAccess: true,
      lifetimeReason: reason,
      lifetimeNotes: notes,
      emailVerified: true,
      skipVerification: true,
      createdBy: 'admin',
      createdAt: new Date(),
      onboardingComplete: false,
    });
    
    return { success: true, created: true, user: newUser };
  } else {
    // Toggle lifetime access
    await db.users.update(
      { id: user.id },
      { 
        lifetimeAccess: true,
        lifetimeReason: reason,
        lifetimeNotes: notes,
        lifetimeGrantedAt: new Date()
      }
    );
    
    return { success: true, created: false, user };
  }
}
```
