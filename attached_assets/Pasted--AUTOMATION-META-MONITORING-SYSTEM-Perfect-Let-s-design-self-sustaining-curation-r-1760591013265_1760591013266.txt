  id SERIAL PRIMARY KEY,

  -- Pattern detection
  request_type TEXT, -- "how_to_do", "how_to_defend", "drilling"
  belt_level TEXT,
  gi_preference TEXT,
  
  requested_at TIMESTAMP DEFAULT NOW()
);

CREATE TABLE technique_meta_status (
  id SERIAL PRIMARY KEY,
  technique_name TEXT UNIQUE,
  
  -- Trend signals
  competition_trend_score DECIMAL DEFAULT 0,
  youtube_trend_score DECIMAL DEFAULT 0,
  user_request_score DECIMAL DEFAULT 0,
  
  -- Combined meta score
  overall_meta_score DECIMAL, -- Weighted average
  meta_status TEXT, -- "rising", "hot", "cooling", "stable"
  
  -- Library coverage
  videos_in_library INTEGER DEFAULT 0,
  highest_quality_video_score DECIMAL,
  coverage_adequate BOOLEAN, -- Do we have enough videos on this?
  
  -- Auto-actions
  needs_curation BOOLEAN DEFAULT FALSE,
  curation_priority INTEGER, -- 1-10
  
  last_updated TIMESTAMP DEFAULT NOW()
);
