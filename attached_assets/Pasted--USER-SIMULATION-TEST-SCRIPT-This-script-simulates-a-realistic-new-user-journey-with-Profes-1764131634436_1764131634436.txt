// Helper to send message and capture response
async function chat(userMessage, context = {}) {
  console.log(`\nğŸ‘¤ USER: ${userMessage}\n`);
  conversationLog.push({ role: 'user', message: userMessage, timestamp: new Date() });
  
  const response = await fetch(`${BASE_URL}/api/chat`, {

    headers: { 'Content-Type': 'application/json' },

      userId: TEST_USER_ID,
      message: userMessage,

    })
  });

function sleep(ms) {
  return new Promise(resolve => setTimeout(resolve, ms));

  const timestamp = new Date().toISOString().replace(/[:.]/g, '-');
  const filename = `simulation-log-${timestamp}.json`;
  
  fs.writeFileSync(filename, JSON.stringify({
    metadata: {
      timestamp: new Date().toISOString(),
      totalMessages: log.length,
      userMessages: log.filter(m => m.role === 'user').length,
      aiResponses: log.filter(m => m.role === 'assistant').length
    },
    conversation: log
  }, null, 2));
  
  console.log(`Conversation log saved to: ${filename}`);
  
  // Also save as plain text for easy reading
  const textFilename = `simulation-log-${timestamp}.txt`;
  let textContent = 'BJJ OS - USER SIMULATION TEST LOG\n';
  textContent += '='.repeat(60) + '\n\n';
  
  log.forEach((entry, index) => {
    const role = entry.role === 'user' ? 'ğŸ‘¤ USER' : 'ğŸ¤– PROFESSOR OS';

    textContent += entry.message + '\n\n';
    textContent += '-'.repeat(40) + '\n\n';
  });
  
  fs.writeFileSync(textFilename, textContent);
  console.log(`Plain text log saved to: ${textFilename}`);
}
