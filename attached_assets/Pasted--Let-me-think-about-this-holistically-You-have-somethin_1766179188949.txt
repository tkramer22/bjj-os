```typescript
interface SynthesizedKnowledge {
  topic: string;
  summary: string;
  approaches: {
    name: string;
    instructor: string;
    coreConcept: string;
    keyDetail: string;
    timestamp: string;
    videoUrl: string;
    bestFor: string; // body type, style, etc.
  }[];
  commonMistakes: string[];
  prerequisites: string[];
  chainsTo: string[];
}

async function synthesizeKnowledgeByTopic(
  topic: string, // e.g., "knee_shield_passing"
  userId: string
): Promise<SynthesizedKnowledge> {
  
  // 1. Get all videos related to this topic
  const videos = await db.ai_video_knowledge.findMany({
    where: {
      OR: [
        { techniqueName: { contains: topic } },
        { positionCategory: { contains: topic } },
        { extractedTechniques: { has: topic } }
      ],
      geminiProcessed: true
    }
  });
  
  // 2. Group by instructor approach
  const approaches = videos.map(video => ({
    name: video.approachName || `${video.instructorName}'s Method`,
    instructor: video.instructorName,
    coreConcept: video.keyConcepts?.[0] || video.techniqueName,
    keyDetail: video.instructorTips?.[0] || video.keyTimestamps?.[0]?.description,
    timestamp: video.keyTimestamps?.[0]?.time || "0:00",
    videoUrl: video.videoUrl,
    bestFor: video.bodyTypeNotes || "all body types"
  }));
  
  // 3. Aggregate common mistakes across all videos
  const commonMistakes = [...new Set(
    videos.flatMap(v => v.commonMistakes || [])
  )];
  
  // 4. Aggregate prerequisites
  const prerequisites = [...new Set(
    videos.flatMap(v => v.prerequisites || [])
  )];
  
  // 5. Aggregate technique chains
  const chainsTo = [...new Set(
    videos.flatMap(v => v.chainsTo || [])
  )];
  
  // 6. Generate summary
  const summary = `${approaches.length} different approaches from elite 
    instructors. Main schools: ${approaches.slice(0,2).map(a => 
    `${a.instructor} (${a.coreConcept})`).join(' vs ')}`;
  
  return {
    topic,
    summary,
    approaches,
    commonMistakes,
    prerequisites,
    chainsTo
  };
}
```
