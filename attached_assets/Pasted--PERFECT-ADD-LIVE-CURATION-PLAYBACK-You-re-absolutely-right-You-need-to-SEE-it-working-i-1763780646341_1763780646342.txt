function CurationProgress({ isRunning, updates }) {
  const progressRef = useRef(null);
  
  // Auto-scroll to bottom as new updates arrive
  useEffect(() => {
    if (progressRef.current) {
      progressRef.current.scrollTop = progressRef.current.scrollHeight;
    }
  }, [updates]);
  
  if (!isRunning && updates.length === 0) return null;
  
  return (
    <div className="curation-progress">
      <div className="progress-header">
        <h4>
          {isRunning ? 'â³ Curation Running...' : 'âœ… Curation Complete'}
        </h4>
        {isRunning && <div className="loading-spinner"></div>}
      </div>
      
      <div className="progress-feed" ref={progressRef}>
        {updates.map((update, i) => (
          <div key={i} className={`progress-item ${update.type}`}>
            <span className="timestamp">{update.time}</span>
            <span className="icon">{update.icon}</span>
            <span className="message">{update.message}</span>
            {update.data && (
              <span className="data">{update.data}</span>
            )}
          </div>
        ))}
      </div>
      
      {isRunning && (
        <div className="current-step">
          <div className="progress-bar">
            <div 
              className="progress-fill" 
              style={{ width: `${(updates.length / 20) * 100}%` }}
            />
          </div>
          <p className="step-text">
            Step {updates.length} of ~20
          </p>
        </div>
      )}
    </div>
  );
}
