function buildOptimalSearchPlan(instructors, gaps, maxSearches) {
  const plan = [];
  
  // Prioritize JT Torres (founder's coach - maximum credibility)
  const jtTorres = instructors.find(i => i.instructor.toLowerCase().includes('jt torres'));
  if (jtTorres) {
    // Give JT 20% of daily searches
    const jtSearchCount = Math.floor(maxSearches * 0.2);
    for (const gap of gaps.slice(0, jtSearchCount)) {
      plan.push({
        instructor: jtTorres.instructor,
        technique: gap.primary_technique,
        template: selectBestTemplate(gap),
        priority: 'HIGHEST'
      });
    }
  }
  
  // Distribute remaining searches across other elite instructors
  const remainingSearches = maxSearches - plan.length;
  const otherInstructors = instructors.filter(i => i !== jtTorres);
  
  // Round-robin through instructors and gaps
  let instructorIndex = 0;
  let gapIndex = 0;
  
  while (plan.length < maxSearches && gapIndex < gaps.length) {
    const instructor = otherInstructors[instructorIndex % otherInstructors.length];
    const gap = gaps[gapIndex];
    
    plan.push({
      instructor: instructor.instructor,
      technique: gap.primary_technique,
      template: selectBestTemplate(gap),
      priority: 'HIGH'
    });
    
    instructorIndex++;
    if (instructorIndex % otherInstructors.length === 0) {
      gapIndex++; // Move to next technique after cycling through all instructors
    }
  }
  
  return plan;
}

function selectBestTemplate(gap) {
  // Select template based on current coverage
  const templates = Object.keys(SEARCH_TEMPLATES);
  const index = gap.current_count % templates.length;
  return templates[index]; // Rotate through templates for variety
}

async function searchYouTubeForInstructor(instructor, technique, template) {
  const query = SEARCH_TEMPLATES[template]
    .replace('{instructor}', instructor)
    .replace('{technique}', technique);
  
  const response = await youtube.search.list({
    part: 'snippet',
    q: query,
    type: 'video',
    maxResults: ELITE_CURATION_CONFIG.RESULTS_PER_SEARCH,
    videoDuration: 'medium',  // 4-20 minutes (optimal for instruction)
    relevanceLanguage: 'en',
    order: ELITE_CURATION_CONFIG.PRIORITIZE_RECENT ? 'date' : 'relevance',
    publishedAfter: ELITE_CURATION_CONFIG.PRIORITIZE_RECENT 
      ? new Date(Date.now() - 3 * 365 * 24 * 60 * 60 * 1000).toISOString() 
      : undefined
  });
  
  return response.data.items.map(item => ({
    videoId: item.id.videoId,
    title: item.snippet.title,
    description: item.snippet.description,
    thumbnail: item.snippet.thumbnails.high.url,
    publishedAt: item.snippet.publishedAt
  }));
}
