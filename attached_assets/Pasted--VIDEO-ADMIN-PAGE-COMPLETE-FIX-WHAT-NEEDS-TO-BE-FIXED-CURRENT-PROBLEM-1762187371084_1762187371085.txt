// Get video statistics
app.get('/api/admin/videos/stats', async (req, res) => {
  try {
    const today = new Date().toISOString().split('T')[0];
    
    // Total videos
    const totalVideos = await db.select({ count: sql<number>`count(*)` })
      .from(videos);
    
    // Videos added today
    const addedToday = await db.select({ count: sql<number>`count(*)` })
      .from(videos)
      .where(sql`DATE(created_at) = ${today}`);
    
    // Curation stats for today
    const curationStats = await db.execute(sql`
      SELECT 
        COALESCE(SUM(videos_added), 0) as added,
        COALESCE(SUM(videos_screened), 0) as searched
      FROM curation_runs
      WHERE run_date = ${today}
    `);
    
    res.json({
      totalVideos: Number(totalVideos[0].count),
      addedToday: Number(curationStats.rows[0].added),
      searchedToday: Number(curationStats.rows[0].searched)
    });
    
  } catch (error) {
    console.error('Video stats error:', error);
    res.status(500).json({ error: 'Failed to load stats' });
  }
});

// Get filtered videos
app.get('/api/admin/videos', async (req, res) => {
  try {
    const { search, belt, style, quality } = req.query;
    
    let query = db.select().from(videos);
    
    // Apply search filter
    if (search && search !== '') {
      query = query.where(
        or(
          sql`LOWER(title) LIKE ${`%${search.toLowerCase()}%`}`,
          sql`LOWER(instructor_name) LIKE ${`%${search.toLowerCase()}%`}`
        )
      );
    }
    
    // Apply belt filter
    if (belt && belt !== 'all') {
      query = query.where(eq(videos.beltLevel, belt));
    }
    
    // Apply style filter
    if (style && style !== 'all') {
      if (style === 'both') {
        // No filter for "both"
      } else {
        query = query.where(eq(videos.giType, style));
      }
    }
    
    // Apply quality filter
    if (quality && quality !== 'all') {
      const minQuality = parseFloat(quality.replace('+', ''));
      query = query.where(gte(videos.credibilityScore, minQuality));
    }
    
    // Order by most recent first
    query = query.orderBy(desc(videos.createdAt));
    
    const results = await query;
    
    res.json(results);
    
  } catch (error) {
    console.error('Video fetch error:', error);
    res.status(500).json({ error: 'Failed to load videos' });
  }
});

// Delete video (admin only)
app.delete('/api/admin/videos/:id', async (req, res) => {
  try {
    const { id } = req.params;
    
    await db.delete(videos).where(eq(videos.id, parseInt(id)));
    
    res.json({ success: true, message: 'Video deleted' });
    
  } catch (error) {
    console.error('Video delete error:', error);
    res.status(500).json({ error: 'Failed to delete video' });
  }
});

// Update video (admin only)
app.patch('/api/admin/videos/:id', async (req, res) => {
  try {
    const { id } = req.params;
    const updates = req.body;
    
    await db.update(videos)
      .set(updates)
      .where(eq(videos.id, parseInt(id)));
    
    res.json({ success: true, message: 'Video updated' });
    
  } catch (error) {
    console.error('Video update error:', error);
    res.status(500).json({ error: 'Failed to update video' });
  }
});
