// Create new component: components/AppStoreBanner.jsx

import { useState, useEffect } from 'react';

export default function AppStoreBanner() {
  const [isVisible, setIsVisible] = useState(false);
  const [isDismissed, setIsDismissed] = useState(false);

  useEffect(() => {
    // Only show on mobile devices
    const isMobile = /iPhone|iPad|iPod|Android/i.test(navigator.userAgent);
    const hasAppInstalled = window.navigator.standalone; // iOS PWA check
    const wasDismissed = localStorage.getItem('appBannerDismissed');
    
    if (isMobile && !hasAppInstalled && !wasDismissed) {
      setIsVisible(true);
    }
  }, []);

  const handleDismiss = () => {
    setIsDismissed(true);
    setIsVisible(false);
    localStorage.setItem('appBannerDismissed', 'true');
  };

  if (!isVisible || isDismissed) return null;

  return (
    <div className="app-store-banner">
      <div className="banner-content">
        <img 
          src="/images/bjj-os-icon.png" 
          alt="BJJ OS" 
          className="app-icon"
        />
        <div className="banner-text">
          <strong>BJJ OS</strong>
          <span>Get the app for the best experience</span>
        </div>
        <a 
          href="https://apps.apple.com/us/app/bjj-os/id6757207452"
          className="banner-cta"
        >
          GET
        </a>
        <button 
          onClick={handleDismiss}
          className="banner-close"
          aria-label="Close"
        >
          âœ•
        </button>
      </div>
    </div>
  );
}
