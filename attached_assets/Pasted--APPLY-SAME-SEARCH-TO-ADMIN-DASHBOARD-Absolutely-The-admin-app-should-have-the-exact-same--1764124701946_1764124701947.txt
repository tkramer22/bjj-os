router.get('/api/admin/videos/search', adminAuth, async (req, res) => {
  const { 
    q, 
    instructor, 
    technique, 
    minQuality,
    dateRange,
    limit = 100 
  } = req.query;

  let query = `
    SELECT 
      id, title, instructor, thumbnail, duration, 
      quality_score, techniques, gi_nogi, youtube_id,
      created_at, updated_at
    FROM videos
    WHERE 1=1
  `;
  
  const params = [];
  let paramIndex = 1;

  // Smart search (same as user)
  if (q) {
    query += ` AND (
      to_tsvector('english', title || ' ' || COALESCE(description, '')) 
      @@ plainto_tsquery('english', $${paramIndex})
      OR title ILIKE $${paramIndex + 1}
      OR instructor ILIKE $${paramIndex + 1}
      OR EXISTS (
        SELECT 1 FROM unnest(techniques) t 
        WHERE t ILIKE $${paramIndex + 1}
      )
    )`;
    params.push(q);
    params.push(`%${q}%`);
    paramIndex += 2;
  }

  // Instructor filter (same as user)
  if (instructor) {
    query += ` AND instructor = $${paramIndex}`;
    params.push(instructor);
    paramIndex++;
  }

  // Technique filter (same as user)
  if (technique) {
    query += ` AND $${paramIndex} = ANY(techniques)`;
    params.push(technique);
    paramIndex++;
  }

  // Admin-only: Quality filter
  if (minQuality) {
    query += ` AND quality_score >= $${paramIndex}`;
    params.push(parseFloat(minQuality));
    paramIndex++;
  }

  // Admin-only: Date range filter
  if (dateRange && dateRange !== 'all') {
    const dateFilters = {
      today: "created_at >= CURRENT_DATE",
      week: "created_at >= CURRENT_DATE - INTERVAL '7 days'",
      month: "created_at >= CURRENT_DATE - INTERVAL '30 days'"
    };
    if (dateFilters[dateRange]) {
      query += ` AND ${dateFilters[dateRange]}`;
    }
  }

  query += ` ORDER BY created_at DESC LIMIT $${paramIndex}`;
  params.push(limit);

  const results = await db.query(query, params);

  // Get total count
  const countResult = await db.query(
    'SELECT COUNT(*) FROM videos WHERE quality_score >= 7.0'
  );

  res.json({
    videos: results.rows,
    count: results.rows.length,
    total: parseInt(countResult.rows[0].count)
  });
});

      updated_at = NOW()
    WHERE id = $5
  `, [title, instructor, techniques, gi_nogi, id]);
