```typescript
// server/routes/videos.ts
app.get('/api/videos/search', async (req, res) => {
  const { technique, position, instructor, giType } = req.query;
  
  let whereConditions: any[] = [
    gte(videos.instructorCredibility, 6.0) // Only show curated videos
  ];
  
  if (technique && technique !== 'all') {
    whereConditions.push(eq(videos.techniqueCategory, technique));
  }
  
  if (position && position !== 'all') {
    whereConditions.push(eq(videos.position, position));
  }
  
  if (instructor && instructor !== 'all') {
    whereConditions.push(eq(videos.instructorName, instructor));
  }
  
  if (giType && giType !== 'all') {
    whereConditions.push(eq(videos.giType, giType));
  }
  
  const results = await db.query.videos.findMany({
    where: and(...whereConditions),
    orderBy: [desc(videos.instructorCredibility), desc(videos.createdAt)],
    limit: 100
  });
  
  res.json(results);
});
```
