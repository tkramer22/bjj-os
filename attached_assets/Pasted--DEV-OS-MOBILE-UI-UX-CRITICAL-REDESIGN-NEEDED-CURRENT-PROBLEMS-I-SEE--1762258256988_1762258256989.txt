// server/routes/admin.ts

app.get('/api/admin/quick-metrics', async (req, res) => {
  try {
    const now = new Date();
    const today = now.toISOString().split('T')[0];
    
    // Quick queries optimized for mobile
    const [
      totalUsers,
      usersToday,
      totalVideos,
      videosToday,
      activeSubscriptions,
      mrr,
      lastCurationRun
    ] = await Promise.all([
      db.execute(sql`SELECT COUNT(*) as count FROM users`),
      db.execute(sql`SELECT COUNT(*) as count FROM users WHERE DATE(created_at) = ${today}`),
      db.execute(sql`SELECT COUNT(*) as count FROM videos`),
      db.execute(sql`SELECT COUNT(*) as count FROM videos WHERE DATE(created_at) = ${today}`),
      db.execute(sql`SELECT COUNT(*) as count FROM users WHERE subscription_status = 'active'`),
      db.execute(sql`
        SELECT COALESCE(SUM(
          CASE 
            WHEN subscription_tier = 'monthly' THEN 50
            WHEN subscription_tier = 'yearly' THEN 500/12
            ELSE 0
          END
        ), 0) as mrr
        FROM users WHERE subscription_status = 'active'
      `),
      db.query.curationRuns.findFirst({
        orderBy: [desc(curationRuns.createdAt)],
        limit: 1
      })
    ]);
    
    const minutesSinceRun = lastCurationRun
      ? Math.floor((now.getTime() - new Date(lastCurationRun.createdAt).getTime()) / 60000)
      : null;
    
    res.json({
      totalUsers: Number(totalUsers.rows[0].count),
      signedUpToday: Number(usersToday.rows[0].count),
      totalVideos: Number(totalVideos.rows[0].count),
      videosToday: Number(videosToday.rows[0].count),
      activeSubscriptions: Number(activeSubscriptions.rows[0].count),
      mrr: Number(mrr.rows[0].mrr).toFixed(0),
      curationRunning: minutesSinceRun !== null && minutesSinceRun < 30,
      minutesSinceRun,
      timestamp: now.toISOString()
    });
    
  } catch (error) {
    console.error('Quick metrics error:', error);
    res.status(500).json({ error: 'Failed to fetch metrics' });
  }
});
